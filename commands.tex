\def\@makechapterhead#1{%
	\thispagestyle{style1}
	\vspace*{20\p@}%
	{\parindent \z@ \mycustomraggedright
		\ifnum \c@secnumdepth >\m@ne
		\if@mainmatter
		\bfseries{\Huge \@chapapp}\small\space {\chapternumber\thechapter}
		\par\nobreak
		\vskip 0\p@
		\fi
		\fi
		\interlinepenalty\@M 
		\Huge \bfseries #1\par\nobreak
		\vskip 120\p@
	}
	\newpage}
